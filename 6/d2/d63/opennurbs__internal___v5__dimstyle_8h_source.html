335  class DimstyleField
336  {
337  public:
338  DimstyleField()
339  : m_next(nullptr)
340  , m_field_id(ON_V5x_DimStyle::Field::fn_unset)
341  {
342  m_val.s_val = nullptr;
343  }
344  ~DimstyleField()
345  {
346  if (nullptr != m_next)
347  {
348  delete m_next;
349  m_next = nullptr;
350  }
351  if (nullptr != m_val.s_val)
352  {
353  delete m_val.s_val;
354  m_val.s_val = nullptr;
355  }
356  }
